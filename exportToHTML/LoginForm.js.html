1    import React, {useState, useContext} from "react";
2    import styles from "./LoginForm.module.css";
3    import {useHistory} from "react-router";
4    import {NotificationContext} from "../Contexts/NotificationContext";
5    import {AuthContext} from "../Contexts/AuthContext";
6    
7    export default function LoginForm() {
8      const [isRegistrationActive, setIsRegistrationActive] = useState(false);
9      const [userData, setUserData] = useState({name: "", password: "", password_check: ""});
10     const history = useHistory();
11     const setNotification = useContext(NotificationContext);
12     const setAuthQuery = useContext(AuthContext);
13   
14     async function handleSubmit(e) {
15       e.preventDefault();
16   
17       const actionURL = isRegistrationActive ?
18         `${process.env.HOST_API_URL}?register` :
19         `${process.env.HOST_API_URL}?login`;
20   
21       const response = await fetch(actionURL, {
22         method: "POST",
23         body: JSON.stringify(userData),
24         credentials: "same-origin",
25         headers: {
26           "Content-Type": "application/json"
27         }
28       });
29   
30       const data = await response.json();
31   
32       if (data.err) {
33         setNotification({type: "error", text: data.data, active: true});
34         return;
35       }
36   
37       if (isRegistrationActive) {
38         setIsRegistrationActive(false);
39       } else {
40         setAuthQuery(`${process.env.HOST_API_URL}?isAuthed  `);
41         history.push("/");
42       }
43   
44       setNotification({type: "message", text: data.data, active: true});
45     }
46   
47     function handleInputChange(e) {
48       const target = e.target;
49       const userProperty = target.id;
50       const value = target.value;
51   
52       setUserData(prevState => {
53         return {...prevState, [userProperty]: value};
54       });
55     }
56   
57     return (
58       <form className={styles.login_form}
59             autoComplete={"off"}
60             method={"POST"}>
61         <div className={styles.login_form_group}>
62           <input id={"name"}
63                  type={"text"}
64                  className={styles.login_form_input}
65                  onChange={handleInputChange}
66                  required/>
67           <label className={styles.login_form_label}>Ваш логин</label>
68         </div>
69         <div className={styles.login_form_group}>
70           <input id={"password"}
71                  type={"password"}
72                  className={styles.login_form_input}
73                  onChange={handleInputChange}
74                  required/>
75           <label className={styles.login_form_label}>Ваш пароль</label>
76         </div>
77         {isRegistrationActive ? <div className={styles.login_form_group}>
78           <input id={"password_check"}
79                  type={"password"}
80                  className={styles.login_form_input}
81                  onChange={handleInputChange}
82                  required/>
83           <label className={styles.login_form_label}>Повторите пароль</label>
84         </div> : null}
85         <button className={styles.login_form_submit_button}
86                 type={"button"}
87                 onClick={handleSubmit}>
88           {isRegistrationActive ? "Зарегистрироваться" : "Войти"}
89         </button>
90         <p className={styles.login_form_registration_link}
91            onClick={() => setIsRegistrationActive(!isRegistrationActive)}>
92           {isRegistrationActive ? "Войти в сущетсвующую уч. запись" : "Зарегистрироваться"}
93         </p>
94       </form>
95     );
96   }
